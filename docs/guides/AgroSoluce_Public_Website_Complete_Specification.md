-- Public cooperatives view (read-only)
CREATE VIEW public_cooperatives AS
SELECT 
  id,
  name,
  acronym,
  region,
  department,
  member_count,
  primary_product,
  certifications,
  verification_status,
  created_at
FROM cooperatives
WHERE verification_status = 'verified';

-- Lead capture table
CREATE TABLE demo_requests (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  cooperative_name TEXT NOT NULL,
  contact_name TEXT NOT NULL,
  email TEXT NOT NULL,
  phone TEXT,
  member_count INTEGER,
  message TEXT,
  source TEXT,                       -- 'website', 'toolkit', 'directory'
  utm_source TEXT,
  utm_medium TEXT,
  utm_campaign TEXT,
  status TEXT DEFAULT 'new',         -- 'new', 'contacted', 'demo_scheduled', 'converted'
  created_at TIMESTAMP DEFAULT NOW()
);

-- Toolkit downloads tracking
CREATE TABLE toolkit_downloads (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email TEXT NOT NULL,
  resource_name TEXT NOT NULL,      -- 'eudr_guide', 'child_labor_template', etc.
  downloaded_at TIMESTAMP DEFAULT NOW(),
  converted_to_demo BOOLEAN DEFAULT false,
  converted_to_paid BOOLEAN DEFAULT false
);

CREATE TABLE buyer_registrations (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

  contact_name TEXT NOT NULL,
  email TEXT NOT NULL,
  phone TEXT,
  country TEXT,
  products_interested TEXT[],
  verification_status TEXT DEFAULT 'pending',
  created_at TIMESTAMP DEFAULT NOW()
);
